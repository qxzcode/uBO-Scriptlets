(function() {
    console.log("mathjax-notab v3 loaded");
    if (!window.MathJax) {
        console.log("error: window.MathJax is", window.MathJax);
    } else {
        MathJax.Hub.Config({menuSettings: {inTabOrder: false}});
        console.log("successfully set MathJax config");
    }
})();
